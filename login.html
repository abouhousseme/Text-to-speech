async function loginUser() { const email = document.getElementById("email").value; const password = document.getElementById("password").value; if (!email || !password) { alert("⚠️ البريد الإلكتروني وكلمة المرور مطلوبة!"); return; } try { const response = await fetch('https://from-text-to-voice-6nye.vercel.app/api/login', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ email, password }), }); const result = await response.json(); if (response.ok) { alert(result.message); // رسالة نجاح window.location.href = '/activate.html'; // صفحة التفعيل } else { alert(result.message); // رسالة خطأ } } catch (error) { console.error('❌ حدث خطأ أثناء تسجيل الدخول:', error); alert("⚠️ حدث خطأ أثناء الاتصال بالخادم."); } }
